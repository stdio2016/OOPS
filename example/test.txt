class digit
{
    digit incr()
    {
        return new digit();
    }
    digit next()
    {
        return this;
    }
    void print()
    {
        cout.puts("0");
    }
    void print2()
    {
    }
}

class digit0 : digit
{
    digit _nex;
    digit0(digit d)
    {
        _nex=d;
    }
    digit incr()
    {
        return new digit1(_nex);
    }
    digit next()
    {
        return _nex;
    }
    void print()
    {
        _nex.print2();
        cout.puts("0");
    }
    void print2()
    {
        _nex.print2();
        cout.puts("0");
    }
}

class digit1 : digit
{
    digit _nex;
    digit1(digit d)
    {
        _nex=d;
    }
    digit incr()
    {
        return new digit0(_nex.incr());
    }
    digit next()
    {
        return _nex;
    }
    void print()
    {
        _nex.print2();
        cout.puts("1");
    }
    void print2()
    {
        _nex.print2();
        cout.puts("1");
    }
}
